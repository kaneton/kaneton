\newenvironment{correction}%
   {
     \ifthenelse
	 {
	   \not\boolean{cmode}
	 }
	 {
	   \comment
	 }
	 {
	   \textbf{\color{red}{ ----- correction}}
	 }
   }%
   {
     \ifthenelse
	 {
	   \not\boolean{cmode}
	 }
	 {\endcomment}
	 {
	   \textbf{\color{red}{ ----- /correction}}
	 }
   }
