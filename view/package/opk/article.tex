%
% details
%

\newenvironment{details}
   {
     \ifthenelse
	 {
	   \equal{\mode}{private}
	 }
	 {%
	   \textbf{ \color{red}{[details]}}
	 }
	 {%
	   \comment
	 }
   }
   {
     \ifthenelse
	 {
	   \equal{\mode}{private}
	 }
	 {%
	   \textbf{ \color{red}{[/details]}}
	 }
	 {%
	   \endcomment
	 }
   }
